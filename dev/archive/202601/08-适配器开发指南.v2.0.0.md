## 一、适配器概述

适配器是 AIOS 控制软件的核心组件，负责将 AIOS 标准调用翻译为跨平台的系统操作。

```
AI 请求 → AIOS Daemon → 适配器 → 跨平台 npm 包 / 系统命令
                           ↓
                    [loudness / brightness / nut.js / ...]
```

---

## 二、控制方式优先级

在开发跨平台适配器时，应按以下优先级选择控制方式：

| 优先级 | 方式 | 可靠性 | 适用场景 |
|--------|------|--------|---------|
| P1 | 跨平台 npm 包 | ⭐⭐⭐⭐⭐ | 有成熟包支持的功能 |
| P2 | Node.js 内置 API | ⭐⭐⭐⭐⭐ | 文件系统、进程等 |
| P3 | 系统命令封装 | ⭐⭐⭐⭐ | 平台特定功能 |
| P4 | Playwright | ⭐⭐⭐⭐ | 浏览器/GUI 自动化 |
| P5 | nut.js 模拟输入 | ⭐⭐⭐ | 最后手段 |

## 四、适配器基类

```typescript
// src/base/types.ts
export interface AdapterResult {
  success: boolean;
  data?: Record<string, any>;
  error?: {
    code: string;
    message: string;
  };
}

export interface AdapterCapability {
  id: string;
  name: string;
  description: string;
  permissionLevel: 'public' | 'low' | 'medium' | 'high' | 'critical';
  parameters?: Array<{
    name: string;
    type: string;
    required: boolean;
    description: string;
  }>;
}

// src/base/Adapter.ts
import { AdapterResult, AdapterCapability } from './types';

export abstract class BaseAdapter {
  abstract readonly id: string;
  abstract readonly name: string;
  abstract readonly description: string;
  abstract readonly capabilities: AdapterCapability[];

  /**
   * 初始化适配器
   */
  async initialize(): Promise<void> {
    // 默认无操作，子类可覆盖
  }

  /**
   * 检查适配器是否可用
   */
  abstract checkAvailability(): Promise<boolean>;

  /**
   * 调用能力
   */
  abstract invoke(
    capability: string,
    args: Record<string, any>
  ): Promise<AdapterResult>;

  /**
   * 关闭适配器
   */
  async shutdown(): Promise<void> {
    // 默认无操作，子类可覆盖
  }

  /**
   * 获取当前平台
   */
  protected getPlatform(): 'darwin' | 'win32' | 'linux' {
    return process.platform as 'darwin' | 'win32' | 'linux';
  }

  /**
   * 创建成功结果
   */
  protected success(data?: Record<string, any>): AdapterResult {
    return { success: true, data };
  }

  /**
   * 创建失败结果
   */
  protected failure(code: string, message: string): AdapterResult {
    return { success: false, error: { code, message } };
  }
}
```

## 五、适配器示例：音频控制

```typescript
// src/system/AudioAdapter.ts
import loudness from 'loudness';
import { BaseAdapter, AdapterResult, AdapterCapability } from '../base';

export class AudioAdapter extends BaseAdapter {
  readonly id = 'com.aios.adapter.audio';
  readonly name = 'Audio Controller';
  readonly description = '跨平台音量控制适配器';

  readonly capabilities: AdapterCapability[] = [
    {
      id: 'get_volume',
      name: '获取音量',
      description: '获取当前系统音量',
      permissionLevel: 'public',
    },
    {
      id: 'set_volume',
      name: '设置音量',
      description: '设置系统音量 (0-100)',
      permissionLevel: 'low',
      parameters: [
        { name: 'volume', type: 'number', required: true, description: '音量值 0-100' },
      ],
    },
    {
      id: 'get_muted',
      name: '获取静音状态',
      description: '检查是否静音',
      permissionLevel: 'public',
    },
    {
      id: 'set_muted',
      name: '设置静音',
      description: '设置静音状态',
      permissionLevel: 'low',
      parameters: [
        { name: 'muted', type: 'boolean', required: true, description: '是否静音' },
      ],
    },
    {
      id: 'toggle_mute',
      name: '切换静音',
      description: '切换静音状态',
      permissionLevel: 'low',
    },
  ];

  async checkAvailability(): Promise<boolean> {
    try {
      await loudness.getVolume();
      return true;
    } catch {
      return false;
    }
  }

  async invoke(capability: string, args: Record<string, any>): Promise<AdapterResult> {
    switch (capability) {
      case 'get_volume':
        return this.getVolume();
      case 'set_volume':
        return this.setVolume(args.volume);
      case 'get_muted':
        return this.getMuted();
      case 'set_muted':
        return this.setMuted(args.muted);
      case 'toggle_mute':
        return this.toggleMute();
      default:
        return this.failure('CAPABILITY_NOT_FOUND', `未知能力: ${capability}`);
    }
  }

  private async getVolume(): Promise<AdapterResult> {
    try {
      const volume = await loudness.getVolume();
      return this.success({ volume });
    } catch (error) {
      return this.failure('GET_VOLUME_FAILED', String(error));
    }
  }

  private async setVolume(volume: number): Promise<AdapterResult> {
    try {
      const clamped = Math.max(0, Math.min(100, volume));
      await loudness.setVolume(clamped);
      return this.success({ volume: clamped });
    } catch (error) {
      return this.failure('SET_VOLUME_FAILED', String(error));
    }
  }

  private async getMuted(): Promise<AdapterResult> {
    try {
      const muted = await loudness.getMuted();
      return this.success({ muted });
    } catch (error) {
      return this.failure('GET_MUTED_FAILED', String(error));
    }
  }

  private async setMuted(muted: boolean): Promise<AdapterResult> {
    try {
      await loudness.setMuted(muted);
      return this.success({ muted });
    } catch (error) {
      return this.failure('SET_MUTED_FAILED', String(error));
    }
  }

  private async toggleMute(): Promise<AdapterResult> {
    try {
      const current = await loudness.getMuted();
      await loudness.setMuted(!current);
      return this.success({ muted: !current });
    } catch (error) {
      return this.failure('TOGGLE_MUTE_FAILED', String(error));
    }
  }
}

export const audioAdapter = new AudioAdapter();
```

## 六、适配器示例：浏览器控制

```typescript
// src/browser/BrowserAdapter.ts
import { chromium, Browser, Page } from 'playwright';
import { BaseAdapter, AdapterResult, AdapterCapability } from '../base';

export class BrowserAdapter extends BaseAdapter {
  readonly id = 'com.aios.adapter.browser';
  readonly name = 'Browser Controller';
  readonly description = '跨平台浏览器控制适配器 (Playwright)';

  private browser: Browser | null = null;
  private page: Page | null = null;

  readonly capabilities: AdapterCapability[] = [
    {
      id: 'open_url',
      name: '打开网址',
      description: '在浏览器中打开指定网址',
      permissionLevel: 'low',
      parameters: [
        { name: 'url', type: 'string', required: true, description: '要打开的网址' },
      ],
    },
    {
      id: 'get_current_url',
      name: '获取当前网址',
      description: '获取当前标签页的网址',
      permissionLevel: 'public',
    },
    {
      id: 'search',
      name: '搜索',
      description: '使用搜索引擎搜索',
      permissionLevel: 'low',
      parameters: [
        { name: 'query', type: 'string', required: true, description: '搜索关键词' },
        { name: 'engine', type: 'string', required: false, description: '搜索引擎 (google/bing/duckduckgo)' },
      ],
    },
    {
      id: 'go_back',
      name: '后退',
      description: '返回上一页',
      permissionLevel: 'low',
    },
    {
      id: 'go_forward',
      name: '前进',
      description: '前进到下一页',
      permissionLevel: 'low',
    },
    {
      id: 'new_tab',
      name: '新建标签页',
      description: '打开新标签页',
      permissionLevel: 'low',
      parameters: [
        { name: 'url', type: 'string', required: false, description: '可选，新标签页打开的网址' },
      ],
    },
    {
      id: 'close_tab',
      name: '关闭标签页',
      description: '关闭当前标签页',
      permissionLevel: 'medium',
    },
  ];

  async initialize(): Promise<void> {
    if (!this.browser) {
      this.browser = await chromium.launch({ headless: false });
      const context = await this.browser.newContext();
      this.page = await context.newPage();
    }
  }

  async checkAvailability(): Promise<boolean> {
    try {
      // 检查 Playwright 是否可用
      const { chromium } = await import('playwright');
      return !!chromium;
    } catch {
      return false;
    }
  }

  async invoke(capability: string, args: Record<string, any>): Promise<AdapterResult> {
    // 确保浏览器已启动
    if (!this.browser) {
      await this.initialize();
    }

    switch (capability) {
      case 'open_url':
        return this.openURL(args.url);
      case 'get_current_url':
        return this.getCurrentURL();
      case 'search':
        return this.search(args.query, args.engine);
      case 'go_back':
        return this.goBack();
      case 'go_forward':
        return this.goForward();
      case 'new_tab':
        return this.newTab(args.url);
      case 'close_tab':
        return this.closeTab();
      default:
        return this.failure('CAPABILITY_NOT_FOUND', `未知能力: ${capability}`);
    }
  }

  private async openURL(url: string): Promise<AdapterResult> {
    try {
      await this.page!.goto(url);
      return this.success({ url });
    } catch (error) {
      return this.failure('OPEN_URL_FAILED', String(error));
    }
  }

  private async getCurrentURL(): Promise<AdapterResult> {
    return this.success({ url: this.page?.url() || '' });
  }

  private async search(query: string, engine: string = 'google'): Promise<AdapterResult> {
    const searchUrls: Record<string, string> = {
      google: `https://www.google.com/search?q=${encodeURIComponent(query)}`,
      bing: `https://www.bing.com/search?q=${encodeURIComponent(query)}`,
      duckduckgo: `https://duckduckgo.com/?q=${encodeURIComponent(query)}`,
    };
    const url = searchUrls[engine] || searchUrls.google;
    return this.openURL(url);
  }

  private async goBack(): Promise<AdapterResult> {
    try {
      await this.page?.goBack();
      return this.success();
    } catch (error) {
      return this.failure('GO_BACK_FAILED', String(error));
    }
  }

  private async goForward(): Promise<AdapterResult> {
    try {
      await this.page?.goForward();
      return this.success();
    } catch (error) {
      return this.failure('GO_FORWARD_FAILED', String(error));
    }
  }

  private async newTab(url?: string): Promise<AdapterResult> {
    try {
      const context = this.browser!.contexts()[0];
      this.page = await context.newPage();
      if (url) {
        await this.page.goto(url);
      }
      return this.success();
    } catch (error) {
      return this.failure('NEW_TAB_FAILED', String(error));
    }
  }

  private async closeTab(): Promise<AdapterResult> {
    try {
      await this.page?.close();
      const context = this.browser?.contexts()[0];
      const pages = context?.pages() || [];
      this.page = pages[pages.length - 1] || null;
      return this.success();
    } catch (error) {
      return this.failure('CLOSE_TAB_FAILED', String(error));
    }
  }

  async shutdown(): Promise<void> {
    await this.browser?.close();
    this.browser = null;
    this.page = null;
  }
}

export const browserAdapter = new BrowserAdapter();
```

```typescript
// src/utils/platform.ts
export type Platform = 'darwin' | 'win32' | 'linux';

export function getPlatform(): Platform {
  return process.platform as Platform;
}

export function isMacOS(): boolean {
  return process.platform === 'darwin';
}

export function isWindows(): boolean {
  return process.platform === 'win32';
}

export function isLinux(): boolean {
  return process.platform === 'linux';
}

/**
 * 根据平台选择值
 */
export function platformSelect<T>(options: {
  darwin?: T;
  win32?: T;
  linux?: T;
  default: T;
}): T {
  const platform = getPlatform();
  return options[platform] ?? options.default;
}

// 使用示例
const lockCommand = platformSelect({
  darwin: 'pmset displaysleepnow',
  win32: 'rundll32.exe user32.dll,LockWorkStation',
  linux: 'loginctl lock-session',
  default: '',
});
```

## 九、命令执行工具

```typescript
// src/utils/command.ts
import { exec, spawn } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export interface CommandResult {
  stdout: string;
  stderr: string;
  exitCode: number;
}

/**
 * 执行命令并返回结果
 */
export async function runCommand(command: string): Promise<CommandResult> {
  try {
    const { stdout, stderr } = await execAsync(command);
    return { stdout: stdout.trim(), stderr: stderr.trim(), exitCode: 0 };
  } catch (error: any) {
    return {
      stdout: error.stdout?.trim() || '',
      stderr: error.stderr?.trim() || error.message,
      exitCode: error.code || 1,
    };
  }
}

/**
 * 执行命令（静默模式）
 */
export async function runSilent(command: string): Promise<boolean> {
  const result = await runCommand(command);
  return result.exitCode === 0;
}

/**
 * 根据平台执行不同命令
 */
export async function runPlatformCommand(commands: {
  darwin?: string;
  win32?: string;
  linux?: string;
}): Promise<CommandResult> {
  const platform = process.platform as 'darwin' | 'win32' | 'linux';
  const command = commands[platform];
  
  if (!command) {
    return {
      stdout: '',
      stderr: `Platform ${platform} not supported`,
      exitCode: 1,
    };
  }
  
  return runCommand(command);
}
```

## 十、适配器注册中心

```typescript
// src/registry.ts
import { BaseAdapter } from './base/Adapter';

class AdapterRegistry {
  private adapters: Map<string, BaseAdapter> = new Map();

  /**
   * 注册适配器
   */
  register(adapter: BaseAdapter): void {
    this.adapters.set(adapter.id, adapter);
  }

  /**
   * 获取适配器
   */
  get(id: string): BaseAdapter | undefined {
    return this.adapters.get(id);
  }

  /**
   * 获取所有适配器
   */
  getAll(): BaseAdapter[] {
    return Array.from(this.adapters.values());
  }

  /**
   * 初始化所有适配器
   */
  async initializeAll(): Promise<void> {
    for (const adapter of this.adapters.values()) {
      await adapter.initialize();
    }
  }

  /**
   * 关闭所有适配器
   */
  async shutdownAll(): Promise<void> {
    for (const adapter of this.adapters.values()) {
      await adapter.shutdown();
    }
  }

  /**
   * 查找支持指定能力的适配器
   */
  findByCapability(capabilityId: string): BaseAdapter | undefined {
    for (const adapter of this.adapters.values()) {
      if (adapter.capabilities.some(c => c.id === capabilityId)) {
        return adapter;
      }
    }
    return undefined;
  }
}

export const registry = new AdapterRegistry();

// 注册所有适配器
import { audioAdapter } from './system/AudioAdapter';
import { browserAdapter } from './browser/BrowserAdapter';

registry.register(audioAdapter);
registry.register(browserAdapter);
// ... 注册其他适配器
```

## 十二、开发最佳实践

### 12.1 错误处理

```typescript
// 始终使用 try-catch 包装外部调用
async invoke(capability: string, args: Record<string, any>): Promise<AdapterResult> {
  try {
    // ... 实现逻辑
  } catch (error) {
    // 记录错误日志
    console.error(`[${this.id}] ${capability} failed:`, error);
    
    // 返回友好的错误信息
    return this.failure(
      'OPERATION_FAILED',
      error instanceof Error ? error.message : String(error)
    );
  }
}
```

### 12.2 平台兼容性

