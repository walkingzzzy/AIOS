## 二、开发者工具控制方案

### 2.1 Claude Code

**控制方式**: Anthropic API

```typescript
import Anthropic from '@anthropic-ai/sdk';

class ClaudeCodeAdapter {
  private client: Anthropic;

  constructor(apiKey: string) {
    this.client = new Anthropic({ apiKey });
  }

  async generateCode(prompt: string, language: string): Promise<string> {
    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 4096,
      messages: [{
        role: 'user',
        content: `Generate ${language} code: ${prompt}`
      }]
    });
    return response.content[0].type === 'text' 
      ? response.content[0].text 
      : '';
  }

  async explainCode(code: string): Promise<string> {
    const response = await this.client.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 2048,
      messages: [{
        role: 'user',
        content: `Explain this code:\n\`\`\`\n${code}\n\`\`\``
      }]
    });
    return response.content[0].type === 'text'
      ? response.content[0].text
      : '';
  }
}
```

**npm 包**: `@anthropic-ai/sdk`

### 2.2 VS Code Extension API

**控制方式**: VS Code Extension + Language Server Protocol

```typescript
import * as vscode from 'vscode';

class VSCodeAdapter {
  async openFile(filePath: string): Promise<void> {
    const uri = vscode.Uri.file(filePath);
    await vscode.window.showTextDocument(uri);
  }

  async insertText(text: string): Promise<void> {
    const editor = vscode.window.activeTextEditor;
    if (editor) {
      await editor.edit(editBuilder => {
        editBuilder.insert(editor.selection.active, text);
      });
    }
  }

  async runCommand(command: string): Promise<void> {
    await vscode.commands.executeCommand(command);
  }

  async getActiveFileContent(): Promise<string> {
    const editor = vscode.window.activeTextEditor;
    return editor?.document.getText() || '';
  }
}
```

## 三、设计软件控制方案

### 3.1 Figma

**控制方式**: Figma REST API (跨平台)

```typescript
class FigmaAdapter {
  private token: string;
  private baseUrl = 'https://api.figma.com/v1';

  constructor(token: string) {
    this.token = token;
  }

  async getFile(fileKey: string): Promise<any> {
    const response = await fetch(`${this.baseUrl}/files/${fileKey}`, {
      headers: { 'X-Figma-Token': this.token }
    });
    return response.json();
  }

  async getImages(fileKey: string, nodeIds: string[]): Promise<any> {
    const ids = nodeIds.join(',');
    const response = await fetch(
      `${this.baseUrl}/images/${fileKey}?ids=${ids}&format=png`,
      { headers: { 'X-Figma-Token': this.token } }
    );
    return response.json();
  }

  async getComments(fileKey: string): Promise<any> {
    const response = await fetch(`${this.baseUrl}/files/${fileKey}/comments`, {
      headers: { 'X-Figma-Token': this.token }
    });
    return response.json();
  }

  async postComment(fileKey: string, message: string, position: { x: number; y: number }): Promise<any> {
    const response = await fetch(`${this.baseUrl}/files/${fileKey}/comments`, {
      method: 'POST',
      headers: {
        'X-Figma-Token': this.token,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        message,
        client_meta: { x: position.x, y: position.y }
      })
    });
    return response.json();
  }
}
```

## 四、3D 软件控制方案

### 4.1 Blender

**控制方式**: Python API (bpy) + 命令行

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';
import { existsSync } from 'fs';

const execAsync = promisify(exec);

class BlenderAdapter {
  private blenderPath: string;

  constructor(blenderPath?: string) {
    this.blenderPath = blenderPath || this.detectBlenderPath();
  }

  /**
   * 自动检测 Blender 安装路径
   */
  private detectBlenderPath(): string {
    const platform = process.platform;
    
    // 常见安装路径
    const possiblePaths: Record<string, string[]> = {
      darwin: [
        '/Applications/Blender.app/Contents/MacOS/Blender',
        '/Applications/Blender.app/Contents/MacOS/blender',
      ],
      win32: [
        'C:\\Program Files\\Blender Foundation\\Blender 4.0\\blender.exe',
        'C:\\Program Files\\Blender Foundation\\Blender 3.6\\blender.exe',
        'C:\\Program Files\\Blender Foundation\\Blender\\blender.exe',
      ],
      linux: [
        '/usr/bin/blender',
        '/snap/bin/blender',
        '/usr/local/bin/blender',
      ],
    };

    const paths = possiblePaths[platform] || [];
    for (const path of paths) {
      if (existsSync(path)) {
        return path;
      }
    }

    // 回退到 PATH 中的 blender
    return 'blender';
  }

  /**
   * 检查 Blender 是否可用
   */
  async checkAvailability(): Promise<boolean> {
    try {
      await execAsync(`"${this.blenderPath}" --version`);
      return true;
    } catch {
      return false;
    }
  }

  async runScript(script: string): Promise<string> {
    try {
      // 转义脚本中的特殊字符
      const escapedScript = script.replace(/"/g, '\\"').replace(/\n/g, '\\n');
      const { stdout } = await execAsync(
        `"${this.blenderPath}" --background --python-expr "${escapedScript}"`
      );
      return stdout;
    } catch (error) {
      console.error('[BlenderAdapter] runScript failed:', error);
      throw error;
    }
  }

  async renderScene(blendFile: string, outputPath: string): Promise<void> {
    try {
      await execAsync(
        `"${this.blenderPath}" -b "${blendFile}" -o "${outputPath}" -f 1`
      );
    } catch (error) {
      console.error('[BlenderAdapter] renderScene failed:', error);
      throw error;
    }
  }

  async exportModel(blendFile: string, outputPath: string, format: 'fbx' | 'obj' | 'gltf'): Promise<void> {
    const exportCommands: Record<string, string> = {
      fbx: `bpy.ops.export_scene.fbx(filepath="${outputPath}")`,
      obj: `bpy.ops.export_scene.obj(filepath="${outputPath}")`,
      gltf: `bpy.ops.export_scene.gltf(filepath="${outputPath}")`,
    };

    const script = `
import bpy
bpy.ops.wm.open_mainfile(filepath="${blendFile}")
${exportCommands[format]}
    `.trim();
    
    try {
      await this.runScript(script);
    } catch (error) {
      console.error('[BlenderAdapter] exportModel failed:', error);
      throw error;
    }
  }
}
```

## 五、音视频处理方案

### 5.1 FFmpeg

**控制方式**: CLI (跨平台)

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

class FFmpegAdapter {
  async convertVideo(input: string, output: string, options?: {
    codec?: string;
    bitrate?: string;
    resolution?: string;
  }): Promise<void> {
    let cmd = `ffmpeg -i "${input}"`;
    
    if (options?.codec) cmd += ` -c:v ${options.codec}`;
    if (options?.bitrate) cmd += ` -b:v ${options.bitrate}`;
    if (options?.resolution) cmd += ` -s ${options.resolution}`;
    
    cmd += ` "${output}"`;
    await execAsync(cmd);
  }

  async extractAudio(input: string, output: string): Promise<void> {
    await execAsync(`ffmpeg -i "${input}" -vn -acodec copy "${output}"`);
  }

  async trimVideo(input: string, output: string, start: string, duration: string): Promise<void> {
    await execAsync(
      `ffmpeg -i "${input}" -ss ${start} -t ${duration} -c copy "${output}"`
    );
  }

  async createThumbnail(input: string, output: string, time: string = '00:00:01'): Promise<void> {
    await execAsync(
      `ffmpeg -i "${input}" -ss ${time} -vframes 1 "${output}"`
    );
  }

  async getMediaInfo(input: string): Promise<any> {
    const { stdout } = await execAsync(
      `ffprobe -v quiet -print_format json -show_format -show_streams "${input}"`
    );
    return JSON.parse(stdout);
  }
}
```

**依赖**: 需要安装 FFmpeg CLI
