```

        ┌───────────────┐

        └───────┬───────┘
                ↓ 自然语言
        ┌───────────────┐
        │   AI 引擎     │  ← 三层云端 API (gpt-4o-mini/gemini/claude)

        ┌───────────────┐

        │  (权限+执行)  │

### 2.1 音量控制

**用户说**: "声音太大了" / "调小点" / "静音" / "音量50%"

**实现方式**: `loudness` npm 包（跨平台）

```typescript
import loudness from 'loudness';

class AudioController {
  /**
   * 获取当前音量 (0-100)
   */
  async getVolume(): Promise<number> {
    return await loudness.getVolume();
  }

  /**
   * 设置音量 (0-100)
   */
  async setVolume(volume: number): Promise<void> {
    const clamped = Math.max(0, Math.min(100, volume));
    await loudness.setVolume(clamped);
  }

  /**
   * 获取静音状态
   */
  async getMuted(): Promise<boolean> {
    return await loudness.getMuted();
  }

  /**
   * 设置静音状态
   */
  async setMuted(muted: boolean): Promise<void> {
    await loudness.setMuted(muted);
  }

  /**
   * 切换静音
   */
  async toggleMute(): Promise<boolean> {
    const current = await this.getMuted();
    await this.setMuted(!current);
    return !current;
  }

  /**
   * 增加音量
   */
  async increaseVolume(delta: number = 10): Promise<number> {
    const current = await this.getVolume();
    const newVolume = Math.min(100, current + delta);
    await this.setVolume(newVolume);
    return newVolume;
  }

  /**
   * 减少音量
   */
  async decreaseVolume(delta: number = 10): Promise<number> {
    const current = await this.getVolume();
    const newVolume = Math.max(0, current - delta);
    await this.setVolume(newVolume);
    return newVolume;
  }
}

export const audioController = new AudioController();
```

**平台支持**: Windows ✅ | macOS ✅ | Linux ✅

### 2.4 外观模式 (深色/浅色)

**用户说**: "开深色模式" / "切换到浅色模式"

**实现方式**: `dark-mode` npm 包 + 系统命令

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

class AppearanceController {
  /**
   * 检查当前是否深色模式
   */
  async isDarkMode(): Promise<boolean> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      const { stdout } = await execAsync(
        'defaults read -g AppleInterfaceStyle 2>/dev/null || echo "Light"'
      );
      return stdout.trim() === 'Dark';
    }
    
    if (platform === 'win32') {
      const { stdout } = await execAsync(
        'reg query "HKCU\\Software\\Microsoft\\Windows\\CurrentVersion\\Themes\\Personalize" /v AppsUseLightTheme'
      );
      return stdout.includes('0x0');
    }
    
    if (platform === 'linux') {
      try {
        const { stdout } = await execAsync(
          'gsettings get org.gnome.desktop.interface color-scheme'
        );
        return stdout.includes('dark');
      } catch {
        return false;
      }
    }
    
    return false;
  }

  /**
   * 设置深色/浅色模式
   */
  async setDarkMode(enabled: boolean): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      const script = `osascript -e 'tell app "System Events" to tell appearance preferences to set dark mode to ${enabled}'`;
      await execAsync(script);
    }
    
    if (platform === 'win32') {
      const value = enabled ? 0 : 1;
      await execAsync(
        `reg add "HKCU\\Software\\Microsoft\\Windows\\CurrentVersion\\Themes\\Personalize" /v AppsUseLightTheme /t REG_DWORD /d ${value} /f`
      );
      await execAsync(
        `reg add "HKCU\\Software\\Microsoft\\Windows\\CurrentVersion\\Themes\\Personalize" /v SystemUsesLightTheme /t REG_DWORD /d ${value} /f`
      );
    }
    
    if (platform === 'linux') {
      const scheme = enabled ? 'prefer-dark' : 'prefer-light';
      await execAsync(`gsettings set org.gnome.desktop.interface color-scheme '${scheme}'`);
    }
  }

  /**
   * 切换深色/浅色模式
   */
  async toggleDarkMode(): Promise<boolean> {
    const current = await this.isDarkMode();
    await this.setDarkMode(!current);
    return !current;
  }
}

export const appearanceController = new AppearanceController();
```

**平台支持**: Windows ✅ | macOS ✅ | Linux ✅ (GNOME)

### 2.5 电源管理

**用户说**: "关机" / "重启" / "休眠" / "锁屏"

**实现方式**: 系统命令（跨平台封装）

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

class PowerController {
  /**
   * 锁屏
   */
  async lockScreen(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await execAsync('pmset displaysleepnow');
    } else if (platform === 'win32') {
      await execAsync('rundll32.exe user32.dll,LockWorkStation');
    } else if (platform === 'linux') {
      // 尝试多种锁屏方式
      try {
        await execAsync('loginctl lock-session');
      } catch {
        try {
          await execAsync('gnome-screensaver-command -l');
        } catch {
          await execAsync('xdg-screensaver lock');
        }
      }
    }
  }

  /**
   * 休眠/睡眠
   */
  async sleep(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await execAsync('pmset sleepnow');
    } else if (platform === 'win32') {
      await execAsync('rundll32.exe powrprof.dll,SetSuspendState 0,1,0');
    } else if (platform === 'linux') {
      await execAsync('systemctl suspend');
    }
  }

  /**
   * 关机
   */
  async shutdown(delay: number = 0): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      if (delay > 0) {
        await execAsync(`sudo shutdown -h +${Math.ceil(delay / 60)}`);
      } else {
        await execAsync('osascript -e \'tell app "System Events" to shut down\'');
      }
    } else if (platform === 'win32') {
      await execAsync(`shutdown /s /t ${delay}`);
    } else if (platform === 'linux') {
      if (delay > 0) {
        await execAsync(`shutdown -h +${Math.ceil(delay / 60)}`);
      } else {
        await execAsync('systemctl poweroff');
      }
    }
  }

  /**
   * 重启
   */
  async restart(delay: number = 0): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await execAsync('osascript -e \'tell app "System Events" to restart\'');
    } else if (platform === 'win32') {
      await execAsync(`shutdown /r /t ${delay}`);
    } else if (platform === 'linux') {
      await execAsync('systemctl reboot');
    }
  }

  /**
   * 取消定时关机
   */
  async cancelShutdown(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await execAsync('sudo killall shutdown').catch(() => {});
    } else if (platform === 'win32') {
      await execAsync('shutdown /a');
    } else if (platform === 'linux') {
      await execAsync('shutdown -c').catch(() => {});
    }
  }
}

export const powerController = new PowerController();
```

**平台支持**: Windows ✅ | macOS ✅ | Linux ✅

### 2.6 应用管理

**用户说**: "打开浏览器" / "打开终端" / "关闭 Chrome"

**实现方式**: `open` + `ps-list` npm 包

```typescript
import open from 'open';
import psList from 'ps-list';
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

interface ProcessInfo {
  pid: number;
  name: string;
  cmd?: string;
}

class AppsController {
  /**
   * 常用应用映射表
   */
  private appMapping: Record<string, Record<string, string>> = {
    darwin: {
      '浏览器': 'Safari',
      'safari': 'Safari',
      'chrome': 'Google Chrome',
      '终端': 'Terminal',
      'terminal': 'Terminal',
      '访达': 'Finder',
      'finder': 'Finder',
      '设置': 'System Preferences',
      '邮件': 'Mail',
      '音乐': 'Music',
      '备忘录': 'Notes',
      'vscode': 'Visual Studio Code',
    },
    win32: {
      '浏览器': 'msedge',
      'edge': 'msedge',
      'chrome': 'chrome',
      '终端': 'cmd',
      'terminal': 'wt', // Windows Terminal
      '资源管理器': 'explorer',
      'explorer': 'explorer',
      '设置': 'ms-settings:',
      '邮件': 'outlookmail:',
      '记事本': 'notepad',
      'vscode': 'code',
    },
    linux: {
      '浏览器': 'firefox',
      'firefox': 'firefox',
      'chrome': 'google-chrome',
      '终端': 'gnome-terminal',
      'terminal': 'gnome-terminal',
      '文件管理器': 'nautilus',
      '设置': 'gnome-control-center',
      'vscode': 'code',
    },
  };

  /**
   * 打开应用
   */
  async openApp(appName: string): Promise<void> {
    const platform = process.platform as 'darwin' | 'win32' | 'linux';
    const mapping = this.appMapping[platform] || {};
    const resolvedName = mapping[appName.toLowerCase()] || appName;
    
    if (platform === 'darwin') {
      await execAsync(`open -a "${resolvedName}"`);
    } else if (platform === 'win32') {
      if (resolvedName.includes(':')) {
        // URL scheme
        await open(resolvedName);
      } else {
        await execAsync(`start "" "${resolvedName}"`);
      }
    } else {
      await execAsync(resolvedName);
    }
  }

  /**
   * 打开 URL
   */
  async openURL(url: string): Promise<void> {
    await open(url);
  }

  /**
   * 获取运行中的进程列表
   */
  async getRunningProcesses(): Promise<ProcessInfo[]> {
    const processes = await psList();
    return processes.map(p => ({
      pid: p.pid,
      name: p.name,
      cmd: p.cmd,
    }));
  }

  /**
   * 关闭应用（通过进程名）
   */
  async closeApp(appName: string): Promise<boolean> {
    const platform = process.platform;
    const processes = await this.getRunningProcesses();
    
    const target = processes.find(p => 
      p.name.toLowerCase().includes(appName.toLowerCase())
    );
    
    if (!target) return false;
    
    if (platform === 'win32') {
      await execAsync(`taskkill /PID ${target.pid} /F`);
    } else {
      await execAsync(`kill ${target.pid}`);
    }
    
    return true;
  }

  /**
   * 强制关闭应用
   */
  async forceCloseApp(appName: string): Promise<boolean> {
    const platform = process.platform;
    
    if (platform === 'win32') {
      await execAsync(`taskkill /IM "${appName}*" /F`);
    } else {
      await execAsync(`pkill -9 -f "${appName}"`);
    }
    
    return true;
  }
}

export const appsController = new AppsController();
```

**平台支持**: Windows ✅ | macOS ✅ | Linux ✅

### 2.7 截图

**用户说**: "截个图" / "截取当前窗口"

**实现方式**: `screenshot-desktop` npm 包 + 系统命令

```typescript
import screenshot from 'screenshot-desktop';
import { exec } from 'child_process';
import { promisify } from 'util';
import * as fs from 'fs/promises';
import * as path from 'path';
import * as os from 'os';

const execAsync = promisify(exec);

type ScreenshotType = 'fullscreen' | 'selection' | 'window';

class ScreenshotController {
  /**
   * 生成截图文件名
   */
  private generateFilename(): string {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    return `screenshot-${timestamp}.png`;
  }

  /**
   * 获取默认保存路径
   */
  private getDefaultPath(): string {
    const platform = process.platform;
    if (platform === 'darwin') {
      return path.join(os.homedir(), 'Desktop');
    }
    return path.join(os.homedir(), 'Pictures');
  }

  /**
   * 全屏截图
   */
  async captureFullscreen(savePath?: string): Promise<string> {
    const filename = this.generateFilename();
    const fullPath = savePath || path.join(this.getDefaultPath(), filename);
    
    const imgBuffer = await screenshot({ format: 'png' });
    await fs.writeFile(fullPath, imgBuffer);
    
    return fullPath;
  }

  /**
   * 区域截图（交互式）
   */
  async captureSelection(savePath?: string): Promise<string> {
    const platform = process.platform;
    const filename = this.generateFilename();
    const fullPath = savePath || path.join(this.getDefaultPath(), filename);
    
    if (platform === 'darwin') {
      await execAsync(`screencapture -i "${fullPath}"`);
    } else if (platform === 'win32') {
      // Windows 使用 Snipping Tool
      await execAsync('snippingtool /clip');
      // 需要用户手动保存
    } else if (platform === 'linux') {
      // 使用 gnome-screenshot 或 scrot
      try {
        await execAsync(`gnome-screenshot -a -f "${fullPath}"`);
      } catch {
        await execAsync(`scrot -s "${fullPath}"`);
      }
    }
    
    return fullPath;
  }

  /**
   * 窗口截图
   */
  async captureWindow(savePath?: string): Promise<string> {
    const platform = process.platform;
    const filename = this.generateFilename();
    const fullPath = savePath || path.join(this.getDefaultPath(), filename);
    
    if (platform === 'darwin') {
      await execAsync(`screencapture -w "${fullPath}"`);
    } else if (platform === 'win32') {
      // Alt+PrintScreen 截取活动窗口
      await execAsync(`powershell -command "Add-Type -AssemblyName System.Windows.Forms; [System.Windows.Forms.Screen]::PrimaryScreen | ForEach-Object { $bitmap = New-Object System.Drawing.Bitmap($_.Bounds.Width, $_.Bounds.Height); $graphics = [System.Drawing.Graphics]::FromImage($bitmap); $graphics.CopyFromScreen($_.Bounds.Location, [System.Drawing.Point]::Empty, $_.Bounds.Size); $bitmap.Save('${fullPath}') }"`);
    } else if (platform === 'linux') {
      try {
        await execAsync(`gnome-screenshot -w -f "${fullPath}"`);
      } catch {
        await execAsync(`scrot -u "${fullPath}"`);
      }
    }
    
    return fullPath;
  }

  /**
   * 通用截图方法
   */
  async takeScreenshot(
    type: ScreenshotType = 'fullscreen',
    savePath?: string
  ): Promise<string> {
    switch (type) {
      case 'fullscreen':
        return this.captureFullscreen(savePath);
      case 'selection':
        return this.captureSelection(savePath);
      case 'window':
        return this.captureWindow(savePath);
      default:
        return this.captureFullscreen(savePath);
    }
  }
}

export const screenshotController = new ScreenshotController();
```

**平台支持**: Windows ✅ | macOS ✅ | Linux ✅

### 2.8 窗口管理

**用户说**: "把窗口放到左边" / "最大化窗口"

**实现方式**: `@nut-tree/nut-js` npm 包（跨平台桌面自动化）

```typescript
import { 
  screen, 
  mouse, 
  keyboard, 
  Key,
  Region,
  getActiveWindow,
  getWindows
} from '@nut-tree/nut-js';

interface WindowInfo {
  title: string;
  region: Region;
}

class WindowController {
  /**
   * 获取屏幕尺寸
   */
  async getScreenSize(): Promise<{ width: number; height: number }> {
    const screenWidth = await screen.width();
    const screenHeight = await screen.height();
    return { width: screenWidth, height: screenHeight };
  }

  /**
   * 获取当前活动窗口
   */
  async getActiveWindow(): Promise<WindowInfo | null> {
    try {
      const win = await getActiveWindow();
      const region = await win.region;
      const title = await win.title;
      return { title, region };
    } catch {
      return null;
    }
  }

  /**
   * 获取所有窗口
   */
  async getAllWindows(): Promise<WindowInfo[]> {
    const windows = await getWindows();
    const result: WindowInfo[] = [];
    
    for (const win of windows) {
      try {
        const region = await win.region;
        const title = await win.title;
        result.push({ title, region });
      } catch {
        // 跳过无法访问的窗口
      }
    }
    
    return result;
  }

  /**
   * 窗口左半屏 (使用快捷键)
   */
  async moveWindowToLeftHalf(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      // macOS: 需要第三方工具如 Rectangle，或使用 AppleScript
      // 这里使用通用的键盘快捷键方式
      await keyboard.pressKey(Key.LeftControl, Key.LeftAlt, Key.Left);
      await keyboard.releaseKey(Key.LeftControl, Key.LeftAlt, Key.Left);
    } else if (platform === 'win32') {
      // Windows: Win + Left
      await keyboard.pressKey(Key.LeftSuper, Key.Left);
      await keyboard.releaseKey(Key.LeftSuper, Key.Left);
    } else if (platform === 'linux') {
      // Linux (GNOME): Super + Left
      await keyboard.pressKey(Key.LeftSuper, Key.Left);
      await keyboard.releaseKey(Key.LeftSuper, Key.Left);
    }
  }

  /**
   * 窗口右半屏
   */
  async moveWindowToRightHalf(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await keyboard.pressKey(Key.LeftControl, Key.LeftAlt, Key.Right);
      await keyboard.releaseKey(Key.LeftControl, Key.LeftAlt, Key.Right);
    } else if (platform === 'win32') {
      await keyboard.pressKey(Key.LeftSuper, Key.Right);
      await keyboard.releaseKey(Key.LeftSuper, Key.Right);
    } else if (platform === 'linux') {
      await keyboard.pressKey(Key.LeftSuper, Key.Right);
      await keyboard.releaseKey(Key.LeftSuper, Key.Right);
    }
  }

  /**
   * 最大化窗口
   */
  async maximizeWindow(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      // macOS: Ctrl+Cmd+F 或点击绿色按钮
      await keyboard.pressKey(Key.LeftControl, Key.LeftSuper, Key.F);
      await keyboard.releaseKey(Key.LeftControl, Key.LeftSuper, Key.F);
    } else if (platform === 'win32') {
      // Windows: Win + Up
      await keyboard.pressKey(Key.LeftSuper, Key.Up);
      await keyboard.releaseKey(Key.LeftSuper, Key.Up);
    } else if (platform === 'linux') {
      await keyboard.pressKey(Key.LeftSuper, Key.Up);
      await keyboard.releaseKey(Key.LeftSuper, Key.Up);
    }
  }

  /**
   * 最小化窗口
   */
  async minimizeWindow(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await keyboard.pressKey(Key.LeftSuper, Key.M);
      await keyboard.releaseKey(Key.LeftSuper, Key.M);
    } else if (platform === 'win32') {
      await keyboard.pressKey(Key.LeftSuper, Key.Down);
      await keyboard.releaseKey(Key.LeftSuper, Key.Down);
    } else if (platform === 'linux') {
      await keyboard.pressKey(Key.LeftSuper, Key.H);
      await keyboard.releaseKey(Key.LeftSuper, Key.H);
    }
  }

  /**
   * 关闭当前窗口
   */
  async closeWindow(): Promise<void> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      await keyboard.pressKey(Key.LeftSuper, Key.W);
      await keyboard.releaseKey(Key.LeftSuper, Key.W);
    } else {
      await keyboard.pressKey(Key.LeftAlt, Key.F4);
      await keyboard.releaseKey(Key.LeftAlt, Key.F4);
    }
  }
}

export const windowController = new WindowController();
```

### 2.9 浏览器控制

**实现方式**: `playwright` npm 包（跨平台浏览器自动化）

```typescript
import { chromium, firefox, webkit, Browser, Page } from 'playwright';

type BrowserType = 'chromium' | 'firefox' | 'webkit';

class BrowserController {
  private browser: Browser | null = null;
  private page: Page | null = null;

  /**
   * 启动浏览器
   */
  async launch(browserType: BrowserType = 'chromium'): Promise<void> {
    const browsers = { chromium, firefox, webkit };
    this.browser = await browsers[browserType].launch({ headless: false });
    const context = await this.browser.newContext();
    this.page = await context.newPage();
  }

  /**
   * 打开 URL
   */
  async openURL(url: string): Promise<void> {
    if (!this.page) {
      await this.launch();
    }
    await this.page!.goto(url);
  }

  /**
   * 获取当前页面 URL
   */
  async getCurrentURL(): Promise<string | null> {
    return this.page?.url() || null;
  }

  /**
   * 获取页面标题
   */
  async getPageTitle(): Promise<string | null> {
    return this.page?.title() || null;
  }

  /**
   * 获取页面文本内容
   */
  async getPageText(): Promise<string | null> {
    if (!this.page) return null;
    return await this.page.evaluate(() => document.body.innerText);
  }

  /**
   * 搜索
   */
  async search(query: string, engine: 'google' | 'bing' | 'duckduckgo' = 'google'): Promise<void> {
    const searchUrls = {
      google: `https://www.google.com/search?q=${encodeURIComponent(query)}`,
      bing: `https://www.bing.com/search?q=${encodeURIComponent(query)}`,
      duckduckgo: `https://duckduckgo.com/?q=${encodeURIComponent(query)}`,
    };
    await this.openURL(searchUrls[engine]);
  }

  /**
   * 后退
   */
  async goBack(): Promise<void> {
    await this.page?.goBack();
  }

  /**
   * 前进
   */
  async goForward(): Promise<void> {
    await this.page?.goForward();
  }

  /**
   * 刷新页面
   */
  async refresh(): Promise<void> {
    await this.page?.reload();
  }

  /**
   * 新建标签页
   */
  async newTab(url?: string): Promise<void> {
    if (!this.browser) {
      await this.launch();
    }
    const context = this.browser!.contexts()[0];
    this.page = await context.newPage();
    if (url) {
      await this.page.goto(url);
    }
  }

  /**
   * 关闭当前标签页
   */
  async closeTab(): Promise<void> {
    await this.page?.close();
    const context = this.browser?.contexts()[0];
    const pages = context?.pages() || [];
    this.page = pages[pages.length - 1] || null;
  }

  /**
   * 截取页面截图
   */
  async screenshot(path: string): Promise<void> {
    await this.page?.screenshot({ path, fullPage: true });
  }

  /**
   * 关闭浏览器
   */
  async close(): Promise<void> {
    await this.browser?.close();
    this.browser = null;
    this.page = null;
  }
}

export const browserController = new BrowserController();
```

### 2.10 系统信息

**实现方式**: `systeminformation` npm 包

```typescript
import si from 'systeminformation';

class SystemInfoController {
  /**
   * 获取系统概览
   */
  async getSystemInfo(): Promise<{
    os: string;
    platform: string;
    hostname: string;
    arch: string;
  }> {
    const osInfo = await si.osInfo();
    return {
      os: `${osInfo.distro} ${osInfo.release}`,
      platform: osInfo.platform,
      hostname: osInfo.hostname,
      arch: osInfo.arch,
    };
  }

  /**
   * 获取 CPU 信息
   */
  async getCpuInfo(): Promise<{
    manufacturer: string;
    brand: string;
    cores: number;
    speed: number;
    usage: number;
  }> {
    const cpu = await si.cpu();
    const load = await si.currentLoad();
    return {
      manufacturer: cpu.manufacturer,
      brand: cpu.brand,
      cores: cpu.cores,
      speed: cpu.speed,
      usage: Math.round(load.currentLoad),
    };
  }

  /**
   * 获取内存信息
   */
  async getMemoryInfo(): Promise<{
    total: string;
    used: string;
    free: string;
    usagePercent: number;
  }> {
    const mem = await si.mem();
    const formatBytes = (bytes: number) => {
      const gb = bytes / (1024 * 1024 * 1024);
      return `${gb.toFixed(1)} GB`;
    };
    return {
      total: formatBytes(mem.total),
      used: formatBytes(mem.used),
      free: formatBytes(mem.free),
      usagePercent: Math.round((mem.used / mem.total) * 100),
    };
  }

  /**
   * 获取磁盘信息
   */
  async getDiskInfo(): Promise<Array<{
    mount: string;
    size: string;
    used: string;
    usagePercent: number;
  }>> {
    const disks = await si.fsSize();
    const formatBytes = (bytes: number) => {
      const gb = bytes / (1024 * 1024 * 1024);
      return `${gb.toFixed(1)} GB`;
    };
    return disks.map(disk => ({
      mount: disk.mount,
      size: formatBytes(disk.size),
      used: formatBytes(disk.used),
      usagePercent: Math.round(disk.use),
    }));
  }

  /**
   * 获取电池信息
   */
  async getBatteryInfo(): Promise<{
    hasBattery: boolean;
    percent: number;
    isCharging: boolean;
    timeRemaining: string;
  } | null> {
    const battery = await si.battery();
    if (!battery.hasBattery) {
      return { hasBattery: false, percent: 100, isCharging: false, timeRemaining: 'N/A' };
    }
    return {
      hasBattery: true,
      percent: battery.percent,
      isCharging: battery.isCharging,
      timeRemaining: battery.timeRemaining > 0 
        ? `${Math.floor(battery.timeRemaining / 60)}h ${battery.timeRemaining % 60}m`
        : 'Calculating...',
    };
  }

  /**
   * 获取网络信息
   */
  async getNetworkInfo(): Promise<Array<{
    iface: string;
    ip4: string;
    mac: string;
    type: string;
  }>> {
    const interfaces = await si.networkInterfaces();
    return (interfaces as any[])
      .filter(iface => iface.ip4 && !iface.internal)
      .map(iface => ({
        iface: iface.iface,
        ip4: iface.ip4,
        mac: iface.mac,
        type: iface.type,
      }));
  }
}

export const systemInfoController = new SystemInfoController();
```

## 三、跨平台权限管理

### 3.1 权限检查策略

由于 Node.js 跨平台方案不依赖系统级权限（如 macOS TCC），权限管理主要关注：

| 权限类型 | 说明 | 检查方式 |
|---------|------|---------|
| 文件系统 | 读写文件权限 | try-catch 捕获 EACCES |
| 进程管理 | 结束其他进程 | 需要管理员权限 |
| 系统设置 | 修改系统配置 | 平台特定检查 |
| 屏幕截图 | 截取屏幕 | macOS 需要屏幕录制权限 |

### 3.2 权限检查实现

```typescript
import { exec } from 'child_process';
import { promisify } from 'util';
import * as fs from 'fs/promises';

const execAsync = promisify(exec);

class PermissionManager {
  /**
   * 检查是否有管理员权限
   */
  async isAdmin(): Promise<boolean> {
    const platform = process.platform;
    
    if (platform === 'win32') {
      try {
        await execAsync('net session');
        return true;
      } catch {
        return false;
      }
    } else {
      return process.getuid?.() === 0;
    }
  }

  /**
   * 检查文件写入权限
   */
  async canWriteFile(filePath: string): Promise<boolean> {
    try {
      await fs.access(filePath, fs.constants.W_OK);
      return true;
    } catch {
      return false;
    }
  }

  /**
   * 检查屏幕录制权限 (macOS)
   */
  async hasScreenCapturePermission(): Promise<boolean> {
    if (process.platform !== 'darwin') {
      return true; // 其他平台默认有权限
    }
    
    try {
      // 尝试截图来检测权限
      const screenshot = await import('screenshot-desktop');
      await screenshot.default();
      return true;
    } catch {
      return false;
    }
  }

  /**
   * 请求管理员权限运行命令
   */
  async runAsAdmin(command: string): Promise<string> {
    const platform = process.platform;
    
    if (platform === 'darwin') {
      const { stdout } = await execAsync(
        `osascript -e 'do shell script "${command}" with administrator privileges'`
      );
      return stdout;
    } else if (platform === 'win32') {
      const { stdout } = await execAsync(
        `powershell -Command "Start-Process cmd -ArgumentList '/c ${command}' -Verb RunAs -Wait"`
      );
      return stdout;
    } else {
      const { stdout } = await execAsync(`pkexec ${command}`);
      return stdout;
    }
  }
}

export const permissionManager = new PermissionManager();
```

```typescript
export enum AIOSErrorCode {
  // 通用错误
  UNKNOWN = 'UNKNOWN',
  TIMEOUT = 'TIMEOUT',
  INVALID_REQUEST = 'INVALID_REQUEST',
  INTERNAL_ERROR = 'INTERNAL_ERROR',
  
  // 权限错误
  PERMISSION_DENIED = 'PERMISSION_DENIED',
  ADMIN_REQUIRED = 'ADMIN_REQUIRED',
  AUTHORIZATION_FAILED = 'AUTHORIZATION_FAILED',
  
  // 资源错误
  APP_NOT_FOUND = 'APP_NOT_FOUND',
  FILE_NOT_FOUND = 'FILE_NOT_FOUND',
  DEVICE_NOT_FOUND = 'DEVICE_NOT_FOUND',
  RESOURCE_BUSY = 'RESOURCE_BUSY',
  RESOURCE_UNAVAILABLE = 'RESOURCE_UNAVAILABLE',
  
  // 操作错误
  OPERATION_FAILED = 'OPERATION_FAILED',
  NOT_SUPPORTED = 'NOT_SUPPORTED',
  INVALID_ARGUMENT = 'INVALID_ARGUMENT',
  OPERATION_CANCELLED = 'OPERATION_CANCELLED',
  
  // 平台错误
  PLATFORM_NOT_SUPPORTED = 'PLATFORM_NOT_SUPPORTED',
  DEPENDENCY_MISSING = 'DEPENDENCY_MISSING',
  
  // 网络错误
  NETWORK_ERROR = 'NETWORK_ERROR',
  API_ERROR = 'API_ERROR',
}

export class AIOSError extends Error {
  constructor(
    public code: AIOSErrorCode,
    message: string,
    public details?: Record<string, any>
  ) {
    super(message);
    this.name = 'AIOSError';
  }

  toJSON() {
    return {
      code: this.code,
      message: this.message,
      details: this.details,
    };
  }

  static fromError(error: unknown, defaultCode: AIOSErrorCode = AIOSErrorCode.UNKNOWN): AIOSError {
    if (error instanceof AIOSError) {
      return error;
    }
    if (error instanceof Error) {
      return new AIOSError(defaultCode, error.message);
    }
    return new AIOSError(defaultCode, String(error));
  }
}
