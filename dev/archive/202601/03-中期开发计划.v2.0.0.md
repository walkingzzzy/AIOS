| 播放/暂停 | `PUT /me/player/play` | low |

| 搜索歌曲 | `GET /search` | low |

| 设置音量 | `PUT /me/player/volume` | low |

```typescript
import SpotifyWebApi from 'spotify-web-api-node';

class SpotifyAdapter {
  private api: SpotifyWebApi;

  constructor(clientId: string, clientSecret: string) {
    this.api = new SpotifyWebApi({ clientId, clientSecret });
  }

  async play(deviceId?: string): Promise<void> {
    try {
      await this.api.play({ device_id: deviceId });
    } catch (error) {
      console.error('[SpotifyAdapter] play failed:', error);
      throw error;
    }
  }

  async pause(): Promise<void> {
    try {
      await this.api.pause();
    } catch (error) {
      console.error('[SpotifyAdapter] pause failed:', error);
      throw error;
    }
  }

  async nextTrack(): Promise<void> {
    try {
      await this.api.skipToNext();
    } catch (error) {
      console.error('[SpotifyAdapter] nextTrack failed:', error);
      throw error;
    }
  }

  async previousTrack(): Promise<void> {
    try {
      await this.api.skipToPrevious();
    } catch (error) {
      console.error('[SpotifyAdapter] previousTrack failed:', error);
      throw error;
    }
  }

  async search(query: string): Promise<SpotifyApi.TrackObjectFull[]> {
    try {
      const result = await this.api.searchTracks(query);
      return result.body.tracks?.items || [];
    } catch (error) {
      console.error('[SpotifyAdapter] search failed:', error);
      throw error;
    }
  }

  async getCurrentTrack(): Promise<SpotifyApi.TrackObjectFull | null> {
    try {
      const result = await this.api.getMyCurrentPlayingTrack();
      return result.body.item as SpotifyApi.TrackObjectFull | null;
    } catch (error) {
      console.error('[SpotifyAdapter] getCurrentTrack failed:', error);
      throw error;
    }
  }

  async setVolume(percent: number): Promise<void> {
    try {
      await this.api.setVolume(Math.max(0, Math.min(100, percent)));
    } catch (error) {
      console.error('[SpotifyAdapter] setVolume failed:', error);
      throw error;
    }
  }
}
```

```typescript
      console.error('[DiscordAdapter] sendMessage failed:', error);
      throw error;
    }
  }

    this.client.destroy();
  }
}
```

### 4.1 Gmail

**控制方式**: Gmail API (跨平台)

```typescript
import { google, gmail_v1 } from 'googleapis';
import { OAuth2Client } from 'google-auth-library';

class GmailAdapter {
  private gmail: gmail_v1.Gmail;

  constructor(auth: OAuth2Client) {
    this.gmail = google.gmail({ version: 'v1', auth });
  }

  async sendEmail(to: string, subject: string, body: string): Promise<string> {
    try {
      const message = [
        `To: ${to}`,
        `Subject: ${subject}`,
        '',
        body
      ].join('\n');

      const encodedMessage = Buffer.from(message)
        .toString('base64')
        .replace(/\+/g, '-')
        .replace(/\//g, '_')
        .replace(/=+$/, '');

      const result = await this.gmail.users.messages.send({
        userId: 'me',
        requestBody: { raw: encodedMessage }
      });
      
      return result.data.id || '';
    } catch (error) {
      console.error('[GmailAdapter] sendEmail failed:', error);
      throw error;
    }
  }

  async listMessages(maxResults: number = 10): Promise<gmail_v1.Schema$Message[]> {
    try {
      const result = await this.gmail.users.messages.list({
        userId: 'me',
        maxResults
      });
      return result.data.messages || [];
    } catch (error) {
      console.error('[GmailAdapter] listMessages failed:', error);
      throw error;
    }
  }
}
```

### 7.2 AIOS 作为 MCP Client

**目标**: 让 AIOS 能够调用现有的 MCP 服务器生态

```typescript
import { Client } from '@modelcontextprotocol/sdk/client/index.js';
import { StdioClientTransport } from '@modelcontextprotocol/sdk/client/stdio.js';

class AIOSMCPClient {
  private clients: Map<string, Client> = new Map();

  async connectServer(
    serverId: string,
    command: string,
    args: string[]
  ): Promise<void> {
    const transport = new StdioClientTransport({ command, args });
    const client = new Client({ name: 'aios', version: '1.0.0' }, {});
    await client.connect(transport);
    this.clients.set(serverId, client);
  }

  async listTools(serverId: string): Promise<any[]> {
    const client = this.clients.get(serverId);
    if (!client) throw new Error('Server not connected');
    const result = await client.listTools();
    return result.tools;
  }

  async callTool(
    serverId: string,
    toolName: string,
    args: Record<string, any>
  ): Promise<any> {
    const client = this.clients.get(serverId);
    if (!client) throw new Error('Server not connected');
    return await client.callTool({ name: toolName, arguments: args });
  }

  async disconnect(serverId: string): Promise<void> {
    const client = this.clients.get(serverId);
    if (client) {
      await client.close();
      this.clients.delete(serverId);
    }
  }
}
```

### 7.3 AIOS 作为 MCP Server

**目标**: 让其他 AI 工具能够调用 AIOS 的系统控制能力

```typescript
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import loudness from 'loudness';
import brightness from 'brightness';

class AIOSMCPServer {
  private server: Server;

  constructor() {
    this.server = new Server(
      { name: 'aios', version: '1.0.0' },
      { capabilities: { tools: {} } }
    );

    this.registerTools();
  }

  private registerTools(): void {
    // 音量控制
    this.server.setRequestHandler('tools/list', async () => ({
      tools: [
        {
          name: 'system_volume',
          description: '控制系统音量',
          inputSchema: {
            type: 'object',
            properties: {
              action: { type: 'string', enum: ['get', 'set', 'mute', 'unmute'] },
              value: { type: 'number', minimum: 0, maximum: 100 }
            },
            required: ['action']
          }
        },
        {
          name: 'system_brightness',
          description: '控制屏幕亮度',
          inputSchema: {
            type: 'object',
            properties: {
              action: { type: 'string', enum: ['get', 'set'] },
              value: { type: 'number', minimum: 0, maximum: 100 }
            },
            required: ['action']
          }
        }
      ]
    }));

    this.server.setRequestHandler('tools/call', async (request) => {
      const { name, arguments: args } = request.params;

      if (name === 'system_volume') {
        switch (args.action) {
          case 'get':
            return { content: [{ type: 'text', text: String(await loudness.getVolume()) }] };
          case 'set':
            await loudness.setVolume(args.value);
            return { content: [{ type: 'text', text: 'Volume set' }] };
          case 'mute':
            await loudness.setMuted(true);
            return { content: [{ type: 'text', text: 'Muted' }] };
          case 'unmute':
            await loudness.setMuted(false);
            return { content: [{ type: 'text', text: 'Unmuted' }] };
        }
      }

      if (name === 'system_brightness') {
        switch (args.action) {
          case 'get':
            return { content: [{ type: 'text', text: String(await brightness.get() * 100) }] };
          case 'set':
            await brightness.set(args.value / 100);
            return { content: [{ type: 'text', text: 'Brightness set' }] };
        }
      }

      throw new Error(`Unknown tool: ${name}`);
    });
  }

  async start(): Promise<void> {
    const transport = new StdioServerTransport();
    await this.server.connect(transport);
  }
}
```

### 8.3 A2A 安全认证方案

**认证机制**: Bearer Token + HTTPS

```typescript
interface A2AAuthConfig {
  // 认证方式
  scheme: 'bearer' | 'oauth2' | 'api_key';
  
  // Token 配置
  tokenSecret: string;
  tokenExpiry: number; // 秒
  
  // 权限映射
  permissions: Map<string, string[]>; // token -> allowed skills
}

interface A2ASecurityMiddleware {
  // 验证请求
  validateRequest(req: Request): Promise<AuthResult>;
  
  // 生成 Token
  generateToken(clientId: string, skills: string[]): string;
  
  // 撤销 Token
  revokeToken(token: string): void;
}

class A2AServer {
  private auth: A2ASecurityMiddleware;

  async handleRequest(req: Request): Promise<Response> {
    // 1. 验证 Authorization header
    const authHeader = req.headers.get('Authorization');
    if (!authHeader?.startsWith('Bearer ')) {
      return new Response(JSON.stringify({
        error: { code: 'UNAUTHORIZED', message: 'Missing or invalid authorization' }
      }), { status: 401 });
    }

    const token = authHeader.replace('Bearer ', '');
    const authResult = await this.auth.validateRequest(req);
    
    if (!authResult.valid) {
      return new Response(JSON.stringify({
        error: { code: 'FORBIDDEN', message: authResult.reason }
      }), { status: 403 });
    }

    // 2. 检查技能权限
    const { skill } = await req.json();
    if (!authResult.allowedSkills.includes(skill)) {
      return new Response(JSON.stringify({
        error: { code: 'SKILL_NOT_ALLOWED', message: `Access to skill '${skill}' denied` }
      }), { status: 403 });
    }

    // 3. 执行任务
    return this.executeTask(req);
  }
}
```

**Token 生成示例**:

```typescript
import jwt from 'jsonwebtoken';

interface TokenPayload {
  clientId: string;
  skills: string[];
  iat: number;
}

interface TokenValidationResult {
  valid: boolean;
  payload?: TokenPayload;
  error?: string;
}

class A2ATokenManager {
  private secret: string;
  private expiry: number;

  constructor(secret: string, expirySeconds: number = 3600) {
    if (!secret || secret.length < 32) {
      throw new Error('JWT secret must be at least 32 characters');
    }
    this.secret = secret;
    this.expiry = expirySeconds;
  }

  generateToken(clientId: string, skills: string[]): string {
    return jwt.sign(
      { clientId, skills, iat: Math.floor(Date.now() / 1000) },
      this.secret,
      { expiresIn: this.expiry }
    );
  }

  validateToken(token: string): TokenValidationResult {
    try {
      const payload = jwt.verify(token, this.secret) as TokenPayload;
      return { valid: true, payload };
    } catch (error) {
      return { 
        valid: false, 
        error: error instanceof Error ? error.message : 'Unknown error' 
      };
    }
  }
}
```

### 8.4 A2A Client 实现

```typescript
interface A2ATask {
  id: string;
  skill: string;
  input: Record<string, any>;
}

interface A2AResult {
  taskId: string;
  status: 'completed' | 'failed' | 'pending';
  output?: any;
  error?: string;
}

class A2AClient {
  private baseUrl: string;

  constructor(agentUrl: string) {
    this.baseUrl = agentUrl;
  }

  async getAgentCard(): Promise<any> {
    const response = await fetch(`${this.baseUrl}/.well-known/agent.json`);
    return response.json();
  }

  async sendTask(task: A2ATask): Promise<A2AResult> {
    const response = await fetch(`${this.baseUrl}/tasks`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(task)
    });
    return response.json();
  }

  async getTaskStatus(taskId: string): Promise<A2AResult> {
    const response = await fetch(`${this.baseUrl}/tasks/${taskId}`);
    return response.json();
  }
}
```
